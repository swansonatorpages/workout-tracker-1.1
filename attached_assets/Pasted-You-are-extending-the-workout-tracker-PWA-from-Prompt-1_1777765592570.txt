sw.js — service worker:
  Cache name: 'workout-tracker-v2'
  On install: cache ['./','./index.html','./sw.js','./manifest.json']
  Fetch strategy: cache-first, fallback to network
  On activate: delete caches not named 'workout-tracker-v2'

manifest.json:
  { "name":"Workout Tracker", "short_name":"Workouts",
    "display":"standalone", "orientation":"portrait-primary",
    "theme_color":"#0b0c10", "background_color":"#0b0c10",
    "start_url":"./" }

Add to index.html <head>:
  <link rel="manifest" href="./manifest.json">
  <meta name="apple-mobile-web-app-capable" content="yes">
  <meta name="apple-mobile-web-app-status-bar-style" content="black">
  <script>
    if('serviceWorker' in navigator)
      navigator.serviceWorker.register('./sw.js');
  </script>
